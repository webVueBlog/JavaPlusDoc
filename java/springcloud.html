    # name: user-gateway # 应用名称

    spring: # Spring Boot配置
  application: # 应用配置
    name: user-gateway # 应用名称
  cloud: # Spring Cloud配置
    sentinel: # Sentinel配置
      eager: false # 是否在应用启动时立即加载所有规则
      datasource: # 数据源配置
        flow: # 流控规则数据源
          nacos: # Nacos数据源
            namespace: ${spring.cloud.nacos.discovery.namespace} # Nacos命名空间
            group-id: ${spring.cloud.nacos.discovery.group} # Nacos分组
            server-addr: ${spring.cloud.nacos.discovery.server-addr} # Nacos服务器地址
            data-id: ${spring.application.name}-flow-rules # 流控规则数据ID
            username: ${spring.cloud.nacos.discovery.username} # Nacos用户名
            password: ${spring.cloud.nacos.discovery.password} # Nacos密码
            rule-type: flow # 规则类型
        degrade: # 降级规则数据源
          nacos: # Nacos数据源
            namespace: ${spring.cloud.nacos.discovery.namespace} # Nacos命名空间
            group-id: ${spring.cloud.nacos.discovery.group} # Nacos分组
            server-addr: ${spring.cloud.nacos.discovery.server-addr} # Nacos服务器地址
            data-id: ${spring.application.name}-degrade-rules # 降级规则数据ID
            username: ${spring.cloud.nacos.discovery.username} # Nacos用户名
            password: ${spring.cloud.nacos.discovery.password} # Nacos密码
            rule-type: degrade # 规则类型
    nacos: # Nacos配置
      discovery: # 服务发现配置
        server-addr: ${NACOS_HOST:120.78.158.199}:${NACOS_PORT:9300} # Nacos服务器地址
        username: nacos # Nacos用户名
        password: nacos_admin # Nacos密码
        namespace: gd #public # Nacos命名空间
        group: gd #DEFAULT_GROUP # Nacos分组
      config: # 配置管理配置
        username: ${spring.cloud.nacos.discovery.username} # Nacos用户名
        password: ${spring.cloud.nacos.discovery.password} # Nacos密码
        server-addr: ${spring.cloud.nacos.discovery.server-addr} # Nacos服务器地址
        file-extension: yml # 配置文件扩展名
        namespace: ${spring.cloud.nacos.discovery.namespace} # Nacos命名空间
        group: ${spring.cloud.nacos.discovery.group} # Nacos分组
        shared-configs[0]: # 共享配置
          data-id: application-${spring.profiles.active}.${spring.cloud.nacos.config.file-extension} # 共享配置数据ID
          namespace: ${spring.cloud.nacos.discovery.namespace} # Nacos命名空间
          group: ${spring.cloud.nacos.discovery.group} # Nacos分组
  profiles: # 配置文件
    active: prod # 激活的配置文件
server: # 服务器配置
  undertow: # Undertow配置
    threads: # 线程配置
      # io线程数
      io: 32
      # 工作线程数
      worker: 400

management: # 管理配置
  endpoints: # 端点配置
    web: # Web端点配置
      exposure: # 暴露端点
        include: '*' # 暴露所有端点

feign: # Feign配置
  sentinel: # Sentinel Feign配置
    enabled: true # 启用Sentinel Feign支持
