    # з¬¬42еӨ©

    # Java зҪ‘з»ңзј–зЁӢ

    зҪ‘з»ңзј–зЁӢжҳҜжҢҮзј–еҶҷиҝҗиЎҢеңЁеӨҡдёӘи®ҫеӨҮпјҲи®Ўз®—жңәпјүзҡ„зЁӢеәҸпјҢиҝҷдәӣи®ҫеӨҮйғҪйҖҡиҝҮзҪ‘з»ңиҝһжҺҘиө·жқҘгҖӮ

    java.net еҢ…дёӯжҸҗдҫӣдәҶдёӨз§Қеёёи§Ғзҡ„зҪ‘з»ңеҚҸи®®зҡ„ж”ҜжҢҒпјҡ

    TCPпјҡTCPпјҲиӢұиҜӯпјҡTransmission Control ProtocolпјҢдј иҫ“жҺ§еҲ¶еҚҸи®®пјү жҳҜдёҖз§Қйқўеҗ‘иҝһжҺҘзҡ„гҖҒеҸҜйқ зҡ„гҖҒеҹәдәҺеӯ—иҠӮжөҒзҡ„дј иҫ“еұӮйҖҡдҝЎеҚҸи®®пјҢTCP еұӮжҳҜдҪҚдәҺ IP еұӮд№ӢдёҠпјҢеә”з”ЁеұӮд№ӢдёӢзҡ„дёӯй—ҙеұӮгҖӮTCP дҝқйҡңдәҶдёӨдёӘеә”з”ЁзЁӢеәҸд№Ӣй—ҙзҡ„еҸҜйқ йҖҡдҝЎгҖӮйҖҡеёёз”ЁдәҺдә’иҒ”зҪ‘еҚҸи®®пјҢиў«з§° TCP / IPгҖӮ

    UDPпјҡUDP пјҲиӢұиҜӯпјҡUser Datagram ProtocolпјҢз”ЁжҲ·ж•°жҚ®жҠҘеҚҸи®®пјүпјҢдҪҚдәҺ OSI жЁЎеһӢзҡ„дј иҫ“еұӮгҖӮдёҖдёӘж— иҝһжҺҘзҡ„еҚҸи®®гҖӮжҸҗдҫӣдәҶеә”з”ЁзЁӢеәҸд№Ӣй—ҙиҰҒеҸ‘йҖҒж•°жҚ®зҡ„ж•°жҚ®жҠҘгҖӮз”ұдәҺUDPзјәд№ҸеҸҜйқ жҖ§дё”еұһдәҺж— иҝһжҺҘеҚҸи®®пјҢжүҖд»Ҙеә”з”ЁзЁӢеәҸйҖҡеёёеҝ…йЎ»е®№и®ёдёҖдәӣдёўеӨұгҖҒй”ҷиҜҜжҲ–йҮҚеӨҚзҡ„ж•°жҚ®еҢ…гҖӮ

    Java зҪ‘з»ңзј–зЁӢе…Ғи®ё Java еә”з”ЁзЁӢеәҸйҖҡиҝҮзҪ‘з»ңдёҺе…¶д»–и®Ўз®—жңәиҝӣиЎҢйҖҡдҝЎгҖӮJava жҸҗдҫӣдәҶдёҖеҘ—дё°еҜҢзҡ„ API жқҘж”ҜжҢҒзҪ‘з»ңзј–зЁӢпјҢеҢ…жӢ¬еҘ—жҺҘеӯ—пјҲSocketпјүгҖҒжңҚеҠЎеҷЁеҘ—жҺҘеӯ—пјҲServerSocketпјүгҖҒURLгҖҒURLConnection зӯүгҖӮ

    # е®һзҺ°еҺҹзҗҶ

    1. еҘ—жҺҘеӯ—пјҲSocketпјүпјҡеҘ—жҺҘеӯ—жҳҜзҪ‘з»ңйҖҡдҝЎзҡ„з«ҜзӮ№пјҢз”ЁдәҺеҸ‘йҖҒе’ҢжҺҘж”¶ж•°жҚ®гҖӮJava жҸҗдҫӣдәҶ java.net.Socket зұ»жқҘиЎЁзӨәе®ўжҲ·з«ҜеҘ—жҺҘеӯ—пјҢjava.net.ServerSocket зұ»жқҘиЎЁзӨәжңҚеҠЎеҷЁеҘ—жҺҘеӯ—гҖӮ

    2. URL е’Ң URLConnectionпјҡURLпјҲз»ҹдёҖиө„жәҗе®ҡдҪҚз¬Ұпјүз”ЁдәҺж ҮиҜҶзҪ‘з»ңдёҠзҡ„иө„жәҗгҖӮJava жҸҗдҫӣдәҶ java.net.URL зұ»жқҘиЎЁзӨә URLпјҢjava.net.URLConnection зұ»жқҘиЎЁзӨәдёҺ URL зҡ„иҝһжҺҘгҖӮ

    # дё»иҰҒж–№жі•

    1. еҲӣе»әе®ўжҲ·з«ҜеҘ—жҺҘеӯ—пјҡ
    2. еҲӣе»әжңҚеҠЎеҷЁеҘ—жҺҘеӯ—пјҡ
    3. еҸ‘йҖҒе’ҢжҺҘж”¶ж•°жҚ®пјҡ
    4. дҪҝз”Ё URL е’Ң URLConnectionпјҡ

    # з”ЁйҖ”

    1. е®ўжҲ·з«Ҝ-жңҚеҠЎеҷЁйҖҡдҝЎпјҡйҖҡиҝҮеҘ—жҺҘеӯ—пјҢеҸҜд»Ҙе®һзҺ°е®ўжҲ·з«Ҝе’ҢжңҚеҠЎеҷЁд№Ӣй—ҙзҡ„йҖҡдҝЎгҖӮдҫӢеҰӮпјҢдёҖдёӘ Web жңҚеҠЎеҷЁеҸҜд»Ҙзӣ‘еҗ¬зү№е®ҡзҡ„з«ҜеҸЈпјҢзӯүеҫ…е®ўжҲ·з«Ҝзҡ„иҝһжҺҘиҜ·жұӮпјҢз„¶еҗҺеӨ„зҗҶиҜ·жұӮе№¶иҝ”еӣһе“Қеә”гҖӮ

    2. иө„жәҗи®ҝй—®пјҡйҖҡиҝҮ URL е’Ң URLConnectionпјҢеҸҜд»Ҙи®ҝй—®зҪ‘з»ңдёҠзҡ„иө„жәҗгҖӮдҫӢеҰӮпјҢеҸҜд»ҘдҪҝз”Ё URLConnection зұ»д»Һ URL дёӢиҪҪж–Үд»¶жҲ–еҸ‘йҖҒ HTTP иҜ·жұӮгҖӮ

    # жіЁж„ҸдәӢйЎ№

    import java.io.*;
import java.net.*;

public class NetworkExample {
    public static void main(String[] args) {
        try {
            // еҲӣе»әе®ўжҲ·з«ҜеҘ—жҺҘеӯ—
            Socket socket = new Socket("localhost", 8080);

            // еҸ‘йҖҒж•°жҚ®
            OutputStream output = socket.getOutputStream();
            PrintWriter writer = new PrintWriter(output, true);
            writer.println("Hello, Server!");

            // жҺҘж”¶ж•°жҚ®
            InputStream input = socket.getInputStream();
            BufferedReader reader = new BufferedReader(new InputStreamReader(input));

            // е…ій—ӯиҝһжҺҘ
            socket.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
