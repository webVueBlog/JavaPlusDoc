            return Response(
                {'error': '无权限操作'}, 
                status=status.HTTP_403_FORBIDDEN
            )
        
        if post.status == 'published':
            post.status = 'draft'
        else:
            post.status = 'published'
        
        post.save()
        
        return Response({
            'message': f'文章状态已更改为{post.get_status_display()}',
            'status': post.status

        """获取当前用户的文章"""
