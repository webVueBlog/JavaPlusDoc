    # StringBuilderе’ҢStringBuffer

    StringBuilder

    # StringBufferе’ҢStringBuilderзҡ„еҢәеҲ«

    з”ұдәҺеӯ—з¬ҰдёІжҳҜдёҚеҸҜеҸҳзҡ„пјҢжүҖд»ҘеҪ“йҒҮеҲ°еӯ—з¬ҰдёІжӢјжҺҘпјҲе°Өе…¶жҳҜдҪҝз”Ё+еҸ·ж“ҚдҪңз¬Ұпјүзҡ„ж—¶еҖҷпјҢе°ұйңҖиҰҒиҖғйҮҸжҖ§иғҪзҡ„й—®йўҳпјҢдҪ дёҚиғҪжҜ«ж— йЎҫиҷ‘ең°з”ҹдә§еӨӘеӨҡ String еҜ№иұЎпјҢеҜ№зҸҚиҙөзҡ„еҶ…еӯҳйҖ жҲҗдёҚеҝ…иҰҒзҡ„еҺӢеҠӣгҖӮ

    Java е°ұи®ҫи®ЎдәҶдёҖдёӘдё“й—Ёз”ЁжқҘи§ЈеҶіжӯӨй—®йўҳзҡ„ StringBuffer зұ»гҖӮ

    з”ұдәҺ StringBuffer ж“ҚдҪңеӯ—з¬ҰдёІзҡ„ж–№жі•еҠ дәҶ synchronized е…ій”®еӯ—иҝӣиЎҢдәҶеҗҢжӯҘпјҢдё»иҰҒжҳҜиҖғиҷ‘еҲ°еӨҡзәҝзЁӢзҺҜеўғдёӢзҡ„е®үе…Ёй—®йўҳпјҢжүҖд»ҘеҰӮжһңеңЁйқһеӨҡзәҝзЁӢзҺҜеўғдёӢпјҢжү§иЎҢж•ҲзҺҮе°ұдјҡжҜ”иҫғдҪҺпјҢеӣ дёәеҠ дәҶжІЎеҝ…иҰҒзҡ„й”ҒгҖӮ

    дәҺжҳҜ Java е°ұз»ҷ StringBuffer вҖңз”ҹдәҶдёӘе…„ејҹвҖқпјҢеҗҚеҸ« StringBuilderпјҢиҜҙпјҢвҖңеӯ©еӯҗпјҢдҪ еҲ«з®ЎзәҝзЁӢе®үе…ЁдәҶпјҢдҪ е°ұеңЁеҚ•зәҝзЁӢзҺҜеўғдёӢдҪҝз”ЁпјҢиҝҷж ·ж•ҲзҺҮдјҡй«ҳеҫ—еӨҡпјҢеҰӮжһңиҰҒеңЁеӨҡзәҝзЁӢзҺҜеўғдёӢдҝ®ж”№еӯ—з¬ҰдёІпјҢдҪ еҲ°ж—¶еҖҷеҸҜд»ҘдҪҝз”Ё ThreadLocal жқҘйҒҝе…ҚеӨҡзәҝзЁӢеҶІзӘҒгҖӮвҖқ

    е®һйҷ…ејҖеҸ‘дёӯпјҢStringBuilder зҡ„дҪҝз”Ёйў‘зҺҮд№ҹжҳҜиҝңй«ҳдәҺ StringBufferпјҢз”ҡиҮіеҸҜд»Ҙиҝҷд№ҲиҜҙпјҢStringBuilder е®Ңе…ЁеҸ–д»ЈдәҶ StringBufferгҖӮ

    # StringBuilderзҡ„дҪҝз”Ё

    javap е’Ңеӯ—иҠӮз Ғ

    еҸҜд»ҘзңӢеҲ° Java зј–иҜ‘еҷЁе°Ҷеӯ—з¬ҰдёІжӢјжҺҘж“ҚдҪңпјҲ+пјүиҪ¬жҚўдёәдәҶ StringBuilder еҜ№иұЎзҡ„ append ж–№жі•пјҢз„¶еҗҺеҶҚи°ғз”Ё StringBuilder еҜ№иұЎзҡ„ toString ж–№жі•иҝ”еӣһжӢјжҺҘеҗҺзҡ„еӯ—з¬ҰдёІгҖӮ

    # StringBuilderзҡ„еҶ…йғЁе®һзҺ°

    еңЁ StringBuilder еҜ№иұЎеҲӣе»әж—¶пјҢдјҡдёә value еҲҶй…ҚдёҖе®ҡзҡ„еҶ…еӯҳз©әй—ҙпјҲеҲқе§Ӣе®№йҮҸ 16пјүпјҢз”ЁдәҺеӯҳеӮЁеӯ—з¬ҰдёІгҖӮ

    /**
 * The value is used for character storage.
 */
char[] value;


/**
 * Constructs a string builder with no characters in it and an
 * initial capacity of 16 characters.
 */
public StringBuilder() {
	super(16);
}

    йҡҸзқҖеӯ—з¬ҰдёІзҡ„жӢјжҺҘпјҢvalue ж•°з»„зҡ„й•ҝеәҰдјҡдёҚж–ӯеўһеҠ пјҢеӣ жӯӨеңЁ StringBuilder еҜ№иұЎзҡ„е®һзҺ°дёӯпјҢvalue ж•°з»„зҡ„й•ҝеәҰжҳҜеҸҜд»ҘеҠЁжҖҒжү©еұ•зҡ„пјҢе°ұеғҸArrayListйӮЈж ·гҖӮ

    е®һйҷ…дёҠжҳҜи°ғз”ЁдәҶ AbstractStringBuilder дёӯзҡ„ append(String str) ж–№жі•гҖӮеңЁ AbstractStringBuilder дёӯпјҢappend(String str) ж–№жі•дјҡжЈҖжҹҘеҪ“еүҚеӯ—з¬ҰеәҸеҲ—дёӯзҡ„еӯ—з¬ҰжҳҜеҗҰеӨҹз”ЁпјҢеҰӮжһңдёҚеӨҹз”ЁеҲҷдјҡиҝӣиЎҢжү©е®№пјҢе№¶е°ҶжҢҮе®ҡеӯ—з¬ҰдёІиҝҪеҠ еҲ°еӯ—з¬ҰеәҸеҲ—зҡ„жң«е°ҫгҖӮ

    append(String str) ж–№жі•е°ҶжҢҮе®ҡеӯ—з¬ҰдёІиҝҪеҠ еҲ°еҪ“еүҚеӯ—з¬ҰеәҸеҲ—дёӯгҖӮеҰӮжһңжҢҮе®ҡеӯ—з¬ҰдёІдёә nullпјҢеҲҷиҝҪеҠ еӯ—з¬ҰдёІ "null"пјӣеҗҰеҲҷдјҡжЈҖжҹҘжҢҮе®ҡеӯ—з¬ҰдёІзҡ„й•ҝеәҰпјҢз„¶еҗҺж №жҚ®еҪ“еүҚеӯ—з¬ҰеәҸеҲ—дёӯзҡ„еӯ—з¬Ұж•°е’ҢжҢҮе®ҡеӯ—з¬ҰдёІзҡ„й•ҝеәҰжқҘеҲӨж–ӯжҳҜеҗҰйңҖиҰҒжү©е®№гҖӮ

    еҰӮжһңйңҖиҰҒжү©е®№пјҢеҲҷдјҡи°ғз”Ё ensureCapacityInternal(int minimumCapacity) ж–№жі•гҖӮжү©е®№д№ӢеҗҺпјҢе°ҶжҢҮе®ҡеӯ—з¬ҰдёІзҡ„еӯ—з¬ҰжӢ·иҙқеҲ°еӯ—з¬ҰеәҸеҲ—дёӯгҖӮ

    # tringBuilderзҡ„ reverse ж–№жі•

    • еҰӮжһңжҢҮе®ҡзҡ„жңҖе°Ҹе®№йҮҸеӨ§дәҺеҪ“еүҚе®№йҮҸпјҢеҲҷж–°е®№йҮҸдёәдёӨеҖҚзҡ„ж—§е®№йҮҸеҠ дёҠ 2гҖӮдёәд»Җд№ҲиҰҒеҠ  2 е‘ўпјҹеҜ№дәҺйқһеёёе°Ҹзҡ„еӯ—з¬ҰдёІпјҲжҜ”еҰӮз©әзҡ„жҲ–еҸӘжңүдёҖдёӘеӯ—з¬Ұзҡ„ StringBuilderпјүпјҢд»…д»…е°Ҷе®№йҮҸеҠ еҖҚеҸҜиғҪд»Қз„¶дёҚи¶ід»Ҙе®№зәіжӣҙеӨҡзҡ„еӯ—з¬ҰгҖӮеңЁиҝҷз§Қжғ…еҶөдёӢпјҢ+ 2 жҸҗдҫӣдәҶдёҖдёӘжңҖе°Ҹзҡ„еўһй•ҝйҮҸпјҢзЎ®дҝқеҚідҪҝеҜ№дәҺеҫҲе°Ҹзҡ„еҲқе§Ӣе®№йҮҸпјҢжү©е®№еҗҺд№ҹиғҪиҮіе°‘ж·»еҠ дёҖдәӣеӯ—з¬ҰиҖҢдёҚйңҖиҰҒз«ӢеҚіеҶҚж¬Ўжү©е®№гҖӮ
    • еҰӮжһңжҢҮе®ҡзҡ„жңҖе°Ҹе®№йҮҸе°ҸдәҺзӯүдәҺеҪ“еүҚе®№йҮҸпјҢеҲҷдёҚдјҡиҝӣиЎҢжү©е®№пјҢзӣҙжҺҘиҝ”еӣһеҪ“еүҚеҜ№иұЎгҖӮ

    еңЁиҝӣиЎҢжү©е®№д№ӢеүҚпјҢensureCapacityInternal(int minimumCapacity) ж–№жі•дјҡе…ҲжЈҖжҹҘеҪ“еүҚеӯ—з¬ҰеәҸеҲ—зҡ„е®№йҮҸжҳҜеҗҰи¶іеӨҹпјҢеҰӮжһңдёҚи¶іе°ұдјҡи°ғз”Ё expandCapacity(int minimumCapacity) ж–№жі•иҝӣиЎҢжү©е®№гҖӮexpandCapacity(int minimumCapacity) ж–№жі•йҰ–е…Ҳи®Ўз®—еҮәж–°е®№йҮҸпјҢз„¶еҗҺдҪҝз”Ё Arrays.copyOf(char[] original, int newLength) ж–№жі•е°ҶеҺҹеӯ—з¬Ұж•°з»„жү©е®№еҲ°ж–°е®№йҮҸзҡ„еӨ§е°ҸгҖӮ
